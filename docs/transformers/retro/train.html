30    def sample(self, prompt: str, sample_len):
31        neighbors_str = []
32        sampled = ''
#

prompt = self.tds.text_to_i(prompt) 

34        for i in range(sample_len):
35            while len(neighbors_str) < len(prompt) // self.chunk_len:
36                off = len(neighbors_str) * self.chunk_len

39
40            src = self.tds.text_to_i(prompt)

43            src = src.to(self.device)

51
52            sampled += self.tds.itos[token.item()]
