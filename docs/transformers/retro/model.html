77class SelfAttention(nn.Module):
#
  • d_model is the number of features in transformer embeddings
  • n_heads is the number of attention heads
  • d_k is the number of features per head
  • is_causal indicates whether this is causal attention (masked)
81    def __init__(self, d_model: int, n_heads: int, d_k: int, is_causal: bool):
# 
88        super().__init__()
89
90        self.is_causal = is_causal
91        self.n_heads = n_heads
92        self.d_k = d_k
#

To scale attentions before softmax by dk​​1​ 

95        self.scale = 1 / math.sqrt(self.d_k)
#

Linear layers for query, key and value heads. 

98        self.query = nn.Linear(d_model, n_heads * d_k)
99        self.key = nn.Linear(d_model, n_heads * d_k)
100        self.value = nn.Linear(d_model, n_heads * d_k)
#

Pre-norm layer. The paper uses RMSNorm instead. 

103        self.norm = nn.LayerNorm(d_model)
104
105        self.softmax = nn.Softmax(dim=-1)
106        self.rotary_pe = RotaryPositionalEmbeddings(self.d_k)
107
108        self.output = nn.Linear(n_heads * d_k, d_model)
# 
110    def mask_attention(self, attn: torch.Tensor):
111        if not self.is_causal:
112            return attn
113
114        mask = torch.tril(attn.new_ones(attn.shape[-2:]))
115        return attn.masked_fill(mask == 0, float('-inf'))
#

h batch, seq, d_model

117    def forward(self, h: torch.Tensor):
# 
122        h_res = h
123        h = self.norm(h)
124
125        mh_shape = (*h.shape[:-1], self.n_heads, self.d_k)
126        q = self.query(h).view(mh_shape)
127        k = self.key(h).view(mh_shape)
128        v = self.value(h).view(mh_shape)
129
130        q = self.rotary_pe(q)
131        k = self.rotary_pe(k)
132
133        attn = torch.einsum('bihd,bjhd->bhij', q, k)
134        attn = attn * self.scale
135
136        attn = self.mask_attention(attn)

139

141
142        h = h.reshape(*h.shape[:-2], -1)

194class ChunkedCrossAttention(nn.Module):
# 
195    def __init__(self, d_model, n_heads, d_k, chunk_len):
196        super().__init__()
197
198        self.chunk_len = chunk_len
199        self.n_heads = n_heads
200        self.d_k = d_k
201        self.scale = 1 / math.sqrt(self.d_k)
202
203        self.query = nn.Linear(d_model, n_heads * d_k)
204        self.key = nn.Linear(d_model, n_heads * d_k)
205        self.value = nn.Linear(d_model, n_heads * d_k)
206
207        self.norm = nn.LayerNorm(d_model)
208
209        self.softmax = nn.Softmax(dim=-1)
210
211        self.output = nn.Linear(n_heads * d_k, d_model)
#

h batch_size, seq, d_model e batch_size, chunks, neighbors, seq, d_model

213    def forward(self, h: torch.Tensor, e: torch.Tensor):
# 
219        batch_size, chunks, neighbors, neighbor_len, d_model = e.shape
220
221        if chunks == 0:
222            return h
223
224        h_res = h
225
226        h = h[:, self.chunk_len - 1:]
227        h = self.norm(h)
228        if h.shape[1] < chunks * self.chunk_len:
229            h = torch.cat((h, h.new_zeros(batch_size, chunks * self.chunk_len - h.shape[1], d_model)), dim=1)
230        h = h.reshape(batch_size, chunks, self.chunk_len, d_model)
231
232        q = self.query(h).view(*h.shape[:-1], self.n_heads, self.d_k)
233        k = self.key(e).view(*e.shape[:-1], self.n_heads, self.d_k)
234        v = self.value(e).view(*e.shape[:-1], self.n_heads, self.d_k)
235
236        attn = torch.einsum('bcihd,bcnjhd->bchinj', q, k)
237        attn = attn * self.scale
238
239        attn = self.softmax(attn.view(*attn.shape[:-2], -1)).view(attn.shape)
240
241        h = torch.einsum("bchinj,bcnjhd->bcihd", attn, v)
242
243        h = h.reshape(batch_size, chunks * self.chunk_len, -1)
244
245        h = self.output(h)
246
247        h = torch.cat((h.new_zeros(batch_size, self.chunk_len - 1, d_model), h), dim=1)
248
249        return h[:, :h_res.shape[1], :] + h_res

252class FeedForward(nn.Module):
# 
253    def __init__(self, d_model: int, d_ff: int):
254        super().__init__()
255
256        self.lin1 = nn.Linear(d_model, d_ff)
257        self.lin2 = nn.Linear(d_ff, d_model)
258
259        self.norm = nn.LayerNorm(d_model)
260        self.act = nn.ReLU()
# 
262    def forward(self, x: torch.Tensor):
263        x_res = x
264        x = self.norm(x)
265        x = self.lin1(x)
266        x = self.act(x)
267        x = self.lin2(x)
268
269        return x + x_res

272class Encoder(nn.Module):
# 
273    def __init__(self, chunk_len: int, n_layers: int, ca_layers: Set[int],
274                 d_model: int, n_heads: int, d_k: int, d_ff: int):
275        super().__init__()
276        self.ca_layers = ca_layers
277        self.chunk_len = chunk_len
278        self.ca = nn.ModuleList([CrossAttention(d_model, n_heads, d_k) for _ in range(len(ca_layers))])
279        self.attn = nn.ModuleList([SelfAttention(d_model, n_heads, d_k, is_causal=False) for _ in range(n_layers)])
280        self.ffw = nn.ModuleList([FeedForward(d_model, d_ff) for _ in range(n_layers)])
281
282        self.norm_h = nn.LayerNorm(d_model)
#

e batch_size, chunks, neighbors, seq, d_model h batch_size, seq, d_model

284    def forward(self, e: torch.Tensor, h: torch.Tensor):
# 
290        batch_size, chunks, neighbors, neighbor_len, d_model = e.shape
291        h_split = h[:, :self.chunk_len * chunks, :].reshape(batch_size, chunks, self.chunk_len, d_model)
292        h_split = self.norm_h(h_split)
293
294        p_ca = 0
295        for p in range(len(self.attn)):
296            e = self.attn[p](e.view(-1, neighbor_len, d_model)).view(e.shape)
297
298            if p in self.ca_layers:
299                e = self.ca[p_ca](e, h_split)
300                p_ca += 1
301
302            e = self.ffw[p](e)
303
304        return e

307class RetroModel(nn.Module):
# 
308    def __init__(self, n_vocab: int, d_model: int, n_layers: int, ca_layers: Set[int], chunk_len: int,
309                 n_heads: int, d_k: int, d_ff: int,
310                 encoder: Encoder):
311        super().__init__()
312
313        self.ca_layers = ca_layers
314        self.emb = nn.Embedding(n_vocab, d_model)
315        self.linear = nn.Linear(d_model, d_model)
316        self.encoder = encoder
317        self.cca = nn.ModuleList(
318            [ChunkedCrossAttention(d_model, n_heads, d_k, chunk_len) for _ in range(len(ca_layers))])
319        self.attn = nn.ModuleList([SelfAttention(d_model, n_heads, d_k, is_causal=True) for _ in range(n_layers)])
320        self.ffw = nn.ModuleList([FeedForward(d_model, d_ff) for _ in range(n_layers)])
321        self.read = nn.Linear(d_model, n_vocab)
322
323        self.norm_e = nn.LayerNorm(d_model)
#

x batch_size, seq e batch_size, chunks, neighbors, seq

325    def forward(self, x: torch.Tensor, ret: torch.Tensor):
# 
331        h = self.emb(x)
332
333        ret_emb = self.emb(ret)
334
335        p_ca = 0
336        for p in range(len(self.attn)):
337            h = self.attn[p](h)
338
339            if self.ca_layers and p == min(self.ca_layers):
340                e = self.encoder(ret_emb, h)
341                e = self.norm_e(e)
342
343            if p in self.ca_layers:
344                h = self.cca[p_ca](h, e)
345                p_ca += 1
346
347            h = self.ffw[p](h)
348
349        return self.read(h)
