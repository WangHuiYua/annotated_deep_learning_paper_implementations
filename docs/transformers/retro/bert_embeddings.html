9class BERTChunkEmbeddings:
# 
10    def __init__(self, device: torch.device):
11        with monit.section('Load BERT tokenizer'):
12            self.tokenizer = BertTokenizer.from_pretrained('bert-base-uncased',
13                                                           cache_dir=str(
14                                                               lab.get_data_path() / 'cache' / 'bert-tokenizer'))
15
16        with monit.section('Load BERT model'):
17            self.model = BertModel.from_pretrained("bert-base-uncased",
18                                                   cache_dir=str(lab.get_data_path() / 'cache' / 'bert-model'))
19
20            self.model.to(device)
21
22        self.device = device
# 
24    def trim_chunk(self, chunk: str):
25        stripped = chunk.strip()
26        parts = stripped.split()
27        stripped = stripped[len(parts[0]):-len(parts[-1])]
28
29        stripped = stripped.strip()
30
31        if not stripped:
32            return chunk
33        else:
34            return stripped
# 
36    def __call__(self, chunks: List[str]):
37        with torch.no_grad():
38            trimmed_chunks = [self.trim_chunk(c) for c in chunks]
39            tokens = self.tokenizer(trimmed_chunks, return_tensors='pt', add_special_tokens=False, padding=True)
40
41            input_ids = tokens['input_ids'].to(self.device)
42            attention_mask = tokens['attention_mask'].to(self.device)
43            output = self.model(input_ids=input_ids,
44                                attention_mask=attention_mask,
45                                token_type_ids=tokens['token_type_ids'].to(self.device))
46
47            state = output['last_hidden_state']
48            emb = (state * attention_mask[:, :, None]).sum(dim=1) / attention_mask[:, :, None].sum(dim=1)
49
50            return emb
