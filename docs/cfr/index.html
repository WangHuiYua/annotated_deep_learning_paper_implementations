Regret Minimization in Games with Incomplete Information

The paper Regret Minimization in Games with Incomplete Information introduces notion counterfactual regret and how minimizing counterfactual regret through self-play can be used to reach Nash equilibrium. The paper uses this technique to solve Texas Hold’em Poker.

We tried to keep our Python implementation easy-to-understand like a tutorial; therefore it is not very efficient. We run it on a very simple imperfect information game called Kuhn poker.

Information Set $I_i$

Information set $I_i \in \mathcal{I}_i$ for player $i$ is a similar to a history $h \in H$ but only contain the actions visible to player $i$. That is, the history $h$ will contain actions/event such as cards dealt to the opposing player while $I_i$ will not have them.

$\mathcal{I}_i$ is known as the information partition of player $i$.

#

Current regret of each action $A(I_i)$,

where,

  • $\sigma^t_i$ is the current strategy of player $i$,
  • $\sigma^t$ is the strategy profile which consists of strategies of all players $\sigma^t_1, \sigma^t_2, …$
  • $\sigma^t_{-i}$ is strategies of all players except $\sigma^t_i$
  • $u_i(\sigma^t)$ is the overall utility of player $i$ where $u_i(h)$ is the terminal utility and $\pi^\sigma(h)$ is the probability of $h$ occurring if players chose actions according to $\sigma$.

Calculate strategy

174    def calculate_strategy(self):
# 
178        regret = {a: max(r, 0) for a, r in self.regret.items()}
179        regret_sum = sum(regret.values())
180        if regret_sum > 0:
181            self.strategy = {a: r / regret_sum for a, r in regret.items()}
182        else:
183            count = len(list(a for a in self.regret))
184            self.strategy = {a: 1 / count for a, r in regret.items()}
#

Get normalized average strategy

186    def get_average_strategy(self):
# 
190        cum_strategy = {a: self.average_strategy.get(a, 0.) for a in self.actions()}
191        strategy_sum = sum(cum_strategy.values())
192        if strategy_sum > 0:
193            return {a: r / strategy_sum for a, r in cum_strategy.items()}
194        else:
195            count = len(list(a for a in cum_strategy))
196            return {a: 1 / count for a, r in cum_strategy.items()}

253    def cfr(self, h: History, i: Player, pi_i: float, pi_neg_i: float) -> float:
254        if h.is_terminal():
255            return h.terminal_utility(i)
256        elif h.is_chance():
257            a = h.sample_chance()
258            return self.cfr(h + a, i, pi_i, pi_neg_i)
259
260        info_set_key = h.info_set_key()
261        if info_set_key not in self.info_sets:
262            self.info_sets[info_set_key] = h.new_info_set()
263        I = self.info_sets[info_set_key]
264        v = 0
265        va = {}
266
267        for a in I.actions():
268            if i == h.player():
269                va[a] = self.cfr(h + a, i, pi_i * I.strategy[a], pi_neg_i)
270            else:
271                va[a] = self.cfr(h + a, i, pi_i, pi_neg_i * I.strategy[a])
272            v = v + I.strategy[a] * va[a]
273
274        v = self.update_get_cfv(h, v, i)
275
276        if h.player() == i:
277            self.update_average_strategy(I, pi_i)
278            self.update_regrets(I, va, v, pi_neg_i)
279
280        return v
# 
282    def solve(self):
283        for _ in self.training_loop:
284            self.update_infosets.clear(self.info_sets)
285
286            for i in range(self.n_players):
287                self.cfr(self.create_new_history(), cast(Player, i), 1, 1)
288
289            self.update_infosets(self.info_sets)
290
291            self.tracker(self.info_sets)
292
293            self.update_get_cfv.train()
294
295        logger.inspect(self.info_sets)

345class InfoSetTracker:
# 
346    def __init__(self):
347        tracker.set_histogram(f'strategy.*')
348        tracker.set_histogram(f'average_strategy.*')
349        tracker.set_histogram(f'regret.*')
350        tracker.set_histogram(f'current_regret.*')
# 
352    def __call__(self, info_sets: Dict[str, InfoSet]):
353        with monit.section("Track"):
354            for I in info_sets.values():
355                avg_strategy = I.get_average_strategy()
356                for a in I.actions():
357                    tracker.add({
358                        f'strategy.{I.key}.{a}': I.strategy[a],
359                        f'average_strategy.{I.key}.{a}': avg_strategy[a],
360                        f'regret.{I.key}.{a}': I.regret[a],
361                        f'current_regret.{I.key}.{a}': I.current_regret[a]
362                    })
