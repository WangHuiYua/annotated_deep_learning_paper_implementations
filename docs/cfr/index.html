44    def calculate_policy(self):
45        regret = {a: max(r, 0) for a, r in self.regret.items()}
46        regret_sum = sum(regret.values())
47        if regret_sum > 0:
48            self.strategy = {a: r / regret_sum for a, r in regret.items()}
49        else:
50            count = len(list(a for a in self.regret))
51            self.strategy = {a: 1 / count for a, r in regret.items()}
52
53    def get_average_strategy(self):
54        cum_strategy = {a: self.average_strategy.get(a, 0.) for a in self.actions()}
55        strategy_sum = sum(cum_strategy.values())
56        if strategy_sum > 0:
57            return {a: r / strategy_sum for a, r in cum_strategy.items()}
58        else:
59            count = len(list(a for a in cum_strategy))
60            return {a: 1 / count for a, r in cum_strategy.items()}
61
62    def clear(self):
63        self.current_regret = {a: 0 for a in self.actions()}
64
65    def update_regrets(self):
66        for k, v in self.current_regret.items():
67            self.regret[k] += v

149class InfoSetTracker:
150    def __init__(self):
151        tracker.set_histogram(f'strategy.*')
152        tracker.set_histogram(f'average_strategy.*')
153        tracker.set_histogram(f'regret.*')
154        tracker.set_histogram(f'current_regret.*')
155
156    def __call__(self, info_sets: Dict[str, InfoSet]):
157        with monit.section("Track"):
158            for I in info_sets.values():
159                avg_strategy = I.get_average_strategy()
160                for a in I.actions():
161                    tracker.add({
162                        f'strategy.{I.key}.{a}': I.strategy[a],
163                        f'average_strategy.{I.key}.{a}': avg_strategy[a],
164                        f'regret.{I.key}.{a}': I.regret[a],
165                        f'current_regret.{I.key}.{a}': I.current_regret[a]
166                    })

169class CFR:
170    update_get_cfv: UpdateAndGetCounterFactualValue
171    track_frequency: int
172    info_sets: Dict[str, InfoSet]
173    is_online_update: bool
174    n_players: int
175    create_new_history: Callable[[], History]
176    epochs: int
177
178    def __init__(self, *,
179                 create_new_history,
180                 epochs,
181                 training_loop: TrainingLoop,
182                 update_regrets: UpdateRegrets,
183                 update_infosets: UpdateInfoSets,
184                 update_average_strategy: UpdateAverageStrategy,
185                 update_get_cfv: UpdateAndGetCounterFactualValue = None,
186                 n_players=2,
187                 track_frequency=10,
188                 save_frequency=10):
189        self.update_average_strategy = update_average_strategy
190        self.training_loop = training_loop
191        if update_get_cfv is None:
192            update_get_cfv = UpdateAndGetCounterFactualValue()
193        self.update_get_cfv = update_get_cfv
194        self.update_regrets = update_regrets
195        self.update_infosets = update_infosets
196        self.save_frequency = save_frequency
197        self.track_frequency = track_frequency
198        self.n_players = n_players
199        self.epochs = epochs
200        self.create_new_history = create_new_history
201        self.info_sets = {}
202        self.tracker = InfoSetTracker()
203
204    def cfr(self, h: History, i: Player, pi_i: float, pi_neg_i: float) -> float:
205        if h.is_terminal():
206            return h.terminal_utility(i)
207        elif h.is_chance():
208            a = h.sample_chance()
209            return self.cfr(h + a, i, pi_i, pi_neg_i)
210
211        info_set_key = h.info_set_key()
212        if info_set_key not in self.info_sets:
213            self.info_sets[info_set_key] = h.new_info_set()
214        I = self.info_sets[info_set_key]
215        v = 0
216        va = {}
217
218        for a in I.actions():
219            if i == h.player():
220                va[a] = self.cfr(h + a, i, pi_i * I.strategy[a], pi_neg_i)
221            else:
222                va[a] = self.cfr(h + a, i, pi_i, pi_neg_i * I.strategy[a])
223            v = v + I.strategy[a] * va[a]
224
225        v = self.update_get_cfv(h, v, i)
226
227        if h.player() == i:
228            self.update_average_strategy(I, pi_i)
229            self.update_regrets(I, va, v, pi_neg_i)
230
231        return v
232
233    def solve(self):
234        for _ in self.training_loop:
235            self.update_infosets.clear(self.info_sets)
236
237            for i in range(self.n_players):
238                self.cfr(self.create_new_history(), cast(Player, i), 1, 1)
239
240            self.update_infosets(self.info_sets)
241
242            self.tracker(self.info_sets)
243
244            self.update_get_cfv.train()
245
246        logger.inspect(self.info_sets)
