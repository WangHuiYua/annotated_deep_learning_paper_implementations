33class History(_History):
34    history: str
35
36    def __init__(self, history: str = ''):
37        self.history = history
38
39    def is_terminal(self):
40        if len(self.history) <= 2:
41            return False
42        if self.history[-1] == 'p':
43            return True
44        if self.history[-2:] == 'bb':
45            return True
46
47        return False
48
49    def _terminal_utility(self) -> float:
50        winner = -1 + 2 * (self.history[0] < self.history[1])
51
52        if self.history[-2:] == 'bp':
53            return 1
54        if self.history[-2:] == 'bb':
55            return winner * 2
56        if self.history[-1] == 'p':
57            return winner
58
59    def terminal_utility(self, i: Player) -> float:
60        if i == PLAYERS[0]:
61            return self._terminal_utility()
62        else:
63            return -1 * self._terminal_utility()
64
65    def is_chance(self) -> bool:
66        return len(self.history) < 2
67
68    def __add__(self, other: Action):
69        return History(self.history + other)
70
71    def player(self) -> Player:
72        return cast(Player, len(self.history) % 2)
73
74    def sample_chance(self) -> Action:
75        while True:
76            r = np.random.randint(len(CHANCES))
77            chance = CHANCES[r]
78            for c in self.history:
79                if c == chance:
80                    chance = None
81                    break
82
83            if chance is not None:
84                return cast(Action, chance)

86    def __repr__(self):
87        return repr(self.history)
