Update global step (number of samples processed) when in training mode

50        if self.mode.is_train:

53        with torch.no_grad():

#

Whether to capture model outputs

57        with self.mode.update(is_log_activations=batch_idx.is_last):
#

Get model outputs.

59            output = self.model(data)
#

Calculate and log loss

62        kl_div_loss = self.kl_div_loss(nn.functional.log_softmax(output / self.temperature, dim=-1),
63                                       nn.functional.log_softmax(large_logits / self.temperature, dim=-1))
64        nll_loss = self.loss_func(output, target)
65        loss = self.kl_div_loss_weight * kl_div_loss + self.nll_loss_weight * nll_loss

Calculate and log accuracy

Train the model

75        if self.mode.is_train:
#

Calculate gradients

77            loss.backward()

Take optimizer step

79            self.optimizer.step()
#

Log the model parameters and gradients on last batch of every epoch

81            if batch_idx.is_last:

Clear the gradients

84            self.optimizer.zero_grad()

Save the tracked metrics
