36class MappingNetwork(nn.Module):
# 
37    def __init__(self, features: int, n_layers: int):
38        super().__init__()
39
40        layers = []
41        for i in range(n_layers):
42            layers.append(EqualizedLinear(features, features, lr_mul=1.))
43            layers.append(nn.LeakyReLU(negative_slope=0.2, inplace=True))
44
45        self.net = nn.Sequential(*layers)
# 
47    def forward(self, x: torch.Tensor):
48        x = F.normalize(x, dim=1)
49        return self.net(x)

52class Discriminator(nn.Module):
# 
53    def __init__(self, n_layers, n_image_features=3, n_features=32, max_features=512):
54        super().__init__()
55
56        self.from_rgb = nn.Sequential(
57            EqualizedConv2d(n_image_features, n_features, 1),
58            nn.LeakyReLU(0.2, True),
59        )
60
61        blocks = []
62        out_features = n_features
63        for i in range(n_layers):
64            in_features = out_features
65            out_features = min(out_features * 2, max_features)
66            blocks.append(DiscriminatorBlock(in_features, out_features))
67
68        self.blocks = nn.Sequential(*blocks)
69
70        self.conv = EqualizedConv2d(out_features, out_features, 4)
71        self.logits = EqualizedLinear(out_features, 1)
# 
73    def __call__(self, x: torch.Tensor):
74        x = self.from_rgb(x)
75        x = self.blocks(x)
76
77        x = self.conv(x)
78        x = x.reshape(x.shape[0], -1)
79        x = self.logits(x)
80        return x

83class DiscriminatorBlock(nn.Module):
# 
84    def __init__(self, in_features, out_features):
85        super().__init__()
86        self.residual = EqualizedConv2d(in_features, out_features, kernel_size=1, stride=2)
87
88        self.block = nn.Sequential(
89            EqualizedConv2d(in_features, in_features, kernel_size=3, padding=1),
90            nn.LeakyReLU(0.2, True),
91            EqualizedConv2d(in_features, out_features, kernel_size=3, padding=1),
92            nn.LeakyReLU(0.2, True),
93        )
94
95        self.down_sample = DownSample(out_features)
96        self.scale = 1 / math.sqrt(2)
# 
98    def forward(self, x):
99        residual = self.residual(x)
100        x = self.block(x)
101        x = self.down_sample(x)
102
103        return (x + residual) * self.scale

106class Generator(nn.Module):
# 
107    def __init__(self, n_layers, d_latent, n_features=32, max_features=512):
108        super().__init__()
109
110        blocks = []
111        in_features = n_features
112        for i in range(n_layers):
113            out_features = in_features
114            in_features = min(in_features * 2, max_features)
115            blocks.append(GeneratorBlock(d_latent, in_features, out_features))
116
117        self.blocks = nn.ModuleList(list(reversed(blocks)))
118
119        self.up_sample = UpSample()
120        self.initial_constant = nn.Parameter(torch.randn((1, in_features, 4, 4)))
# 
122    def forward(self, styles: torch.Tensor, input_noise: torch.Tensor):
123        batch_size = styles.shape[0]
124
125        x = self.initial_constant.expand(batch_size, -1, -1, -1)
126
127        rgb = None
128
129        for i in range(len(self.blocks)):
130            if i != 0:
131                x = self.up_sample(x)
132
133            x, rgb = self.blocks[i](x, rgb, styles[:, i], input_noise)
134
135        return rgb

138class GeneratorBlock(nn.Module):
# 
139    def __init__(self, latent_dim, in_features, out_features):
140        super().__init__()
141        self.to_style1 = EqualizedLinear(latent_dim, in_features)
142        self.to_noise1 = EqualizedLinear(1, out_features)
143        self.conv1 = Conv2dWeightModulate(in_features, out_features, kernel_size=3)
144
145        self.to_style2 = EqualizedLinear(latent_dim, out_features)
146        self.to_noise2 = EqualizedLinear(1, out_features)
147        self.conv2 = Conv2dWeightModulate(out_features, out_features, kernel_size=3)
148
149        self.activation = nn.LeakyReLU(0.2, True)
150        self.to_rgb = ToRGB(latent_dim, out_features)
# 
152    def forward(self, x: torch.Tensor, prev_rgb: torch.Tensor, istyle: torch.Tensor, inoise: torch.Tensor):
153        inoise = inoise[:, :x.shape[2], :x.shape[3], :]
154        noise1 = self.to_noise1(inoise).permute((0, 3, 2, 1))
155        noise2 = self.to_noise2(inoise).permute((0, 3, 2, 1))
156
157        style1 = self.to_style1(istyle)
158        x = self.conv1(x, style1)
159        x = self.activation(x + noise1)
160
161        style2 = self.to_style2(istyle)
162        x = self.conv2(x, style2)
163        x = self.activation(x + noise2)
164
165        rgb = self.to_rgb(x, prev_rgb, istyle)
166        return x, rgb

169class ToRGB(nn.Module):
# 
170    def __init__(self, latent_dim, in_features):
171        super().__init__()
172        self.input_channel = in_features
173        self.to_style = EqualizedLinear(latent_dim, in_features)
174
175        self.conv = Conv2dWeightModulate(in_features, 3, kernel_size=1, demodulate=False)
176        self.up_sample = UpSample()
# 
178    def forward(self, x, prev_rgb, istyle):
179        style = self.to_style(istyle)
180        rgb = self.conv(x, style)
181
182        if prev_rgb is not None:
183            rgb = rgb + self.up_sample(prev_rgb)
184
185        return rgb

188class Conv2dWeightModulate(nn.Module):
# 
189    def __init__(self, in_features, out_features, kernel_size, demodulate=True, lr_mul=1., eps=1e-8):
190        super().__init__()
191        self.filters = out_features
192        self.demodulate = demodulate
193        self.kernel = kernel_size
194        self.padding = (self.kernel - 1) // 2
195
196        he_std = 1 / math.sqrt(in_features * kernel_size * kernel_size)
197        self.weight = nn.Parameter(torch.randn((out_features, in_features, kernel_size, kernel_size)) / lr_mul)
198        self.eps = eps
199
200        self.runtime_coef = lr_mul * he_std
# 
202    def forward(self, x, style):
203        b, c, h, w = x.shape
204
205        w1 = style[:, None, :, None, None]
206        w2 = self.weight[None, :, :, :, :] * self.runtime_coef
207        weights = w2 * (w1 + 1)
208
209        if self.demodulate:
210            d = torch.rsqrt((weights ** 2).sum(dim=(2, 3, 4), keepdim=True) + self.eps)
211            weights = weights * d
212
213        x = x.reshape(1, -1, h, w)
214
215        _, _, *ws = weights.shape
216        weights = weights.reshape(b * self.filters, *ws)
217
218        x = F.conv2d(x, weights, padding=self.padding, groups=b)
219
220        x = x.reshape(-1, self.filters, h, w)
221
222        return x

225class DownSample(nn.Module):
# 
226    def __init__(self, features: int):
227        super().__init__()
228        self.smooth = Smooth()
229        self.conv = EqualizedConv2d(features, features, kernel_size=3, padding=1, stride=2)
# 
231    def forward(self, x: torch.Tensor):
232        return self.conv(self.smooth(x))

235class UpSample(nn.Module):
# 
236    def __init__(self):
237        super().__init__()
238        self.up_sample = nn.Upsample(scale_factor=2, mode='bilinear', align_corners=False)
239        self.smooth = Smooth()
# 
241    def forward(self, x: torch.Tensor):
242        return self.smooth(self.up_sample(x))

245class Smooth(nn.Module):
# 
246    def __init__(self):
247        super().__init__()
248        kernel = [[1, 2, 1],
249                  [2, 4, 2],
250                  [1, 2, 1]]
251        kernel = torch.tensor([[kernel]], dtype=torch.float)
252        kernel /= kernel.sum()
253        self.kernel = nn.Parameter(kernel, requires_grad=False)
# 
255    def forward(self, x: torch.Tensor):
256        b, c, h, w = x.shape
257        x = x.view(-1, 1, h, w)
258
259        x = F.conv2d(x, self.kernel, padding=1)
260
261        return x.view(b, c, h, w)

264class EqualizedLinear(nn.Module):
# 
265    def __init__(self, in_features: int, out_features: int, lr_mul: float = 1.):
266        super().__init__()
267        self.weight = nn.Parameter(torch.randn(out_features, in_features) / lr_mul)
268        self.bias = nn.Parameter(torch.zeros(out_features))
269
270        he_std = 1 / math.sqrt(in_features)
271        self.runtime_coef = lr_mul * he_std
272        self.lr_mul = lr_mul
# 
274    def __call__(self, x: torch.Tensor):
275        return F.linear(x, self.weight * self.runtime_coef, bias=self.bias * self.lr_mul)
# 
278class EqualizedConv2d(nn.Module):
# 
279    def __init__(self, in_features: int, out_features: int,
280                 kernel_size: int, padding: int = 0, stride: int = 1,
281                 lr_mul: float = 1.):
282        super().__init__()
283        self.stride = stride
284        self.padding = padding
285        self.weight = nn.Parameter(torch.randn((out_features, in_features, kernel_size, kernel_size)) / lr_mul)
286        self.bias = nn.Parameter(torch.zeros(out_features))
287
288        he_std = 1 / math.sqrt(in_features * kernel_size * kernel_size)
289        self.runtime_coef = lr_mul * he_std
290        self.lr_mul = lr_mul
# 
292    def forward(self, x: torch.Tensor):
293        return F.conv2d(x, self.weight * self.runtime_coef, bias=self.bias * self.lr_mul,
294                        padding=self.padding, stride=self.stride)

297class PathLengthPenalty(nn.Module):
# 
298    def __init__(self, beta: float):
299        super().__init__()
300        self.beta = beta
301        self.steps = nn.Parameter(torch.tensor(0.), requires_grad=False)
302        self.moving_average = nn.Parameter(torch.tensor(0.), requires_grad=False)
# 
304    def forward(self, styles: torch.Tensor, images: torch.Tensor):
305        device = images.device
306        num_pixels = images.shape[2] * images.shape[3]
307        pl_noise = torch.randn(images.shape, device=device) / math.sqrt(num_pixels)
308        outputs = (images * pl_noise).sum()
309
310        gradients, *_ = torch.autograd.grad(outputs=outputs,
311                                            inputs=styles,
312                                            grad_outputs=torch.ones(outputs.shape, device=device),
313                                            create_graph=True)
314
315        path_lengths = (gradients ** 2).sum(dim=2).mean(dim=1).sqrt()
316
317        if self.steps > 0:
318            a = self.moving_average / (1 - self.beta ** self.steps)
319            loss = torch.mean((path_lengths - a) ** 2)
320        else:
321            loss = path_lengths.new_tensor(0)
322
323        mean = path_lengths.mean().detach()
324
325        if not torch.isnan(mean):
326            self.moving_average.mul_(self.beta).add_(mean, alpha=1 - self.beta)
327            self.steps.add(1.)
328
329        return loss

360class Configs(BaseConfigs):
361    device: torch.device = DeviceConfigs()
362    data_loader: Any

# 
401    def init(self):
402        self.dataset = Dataset(self.image_size)
403        dataloader = torch.utils.data.DataLoader(self.dataset, batch_size=self.batch_size,
404                                                 num_workers=32,
405                                                 shuffle=True, drop_last=True, pin_memory=True)
406        self.loader = cycle(dataloader)
407        self.n_layers = int(math.log2(self.image_size)) - 1
408
409        self.discriminator = Discriminator(self.n_layers - 1).to(self.device)
410        self.generator = Generator(self.n_layers, self.d_latent).to(self.device)
411        self.mapping_network = MappingNetwork(self.d_latent, self.mapping_network_layers).to(self.device)
412        self.path_length_penalty = PathLengthPenalty(0.99).to(self.device)
413
414        self.discriminator_loss = DiscriminatorLoss()
415        self.generator_loss = GeneratorLoss()
416
417        self.discriminator_optimizer = torch.optim.Adam(
418            self.discriminator.parameters(),
419            lr=self.learning_rate, betas=self.betas
420        )
421        self.generator_optimizer = torch.optim.Adam(
422            self.generator.parameters(),
423            lr=self.learning_rate, betas=self.betas
424        )
425        self.mapping_network_optimizer = torch.optim.Adam(
426            self.mapping_network.parameters(),
427            lr=self.mapping_network_learning_rate, betas=self.betas
428        )
429        tracker.set_image("generated", True)
# 
431    def z_noise(self, batch_size):
432        return torch.randn(batch_size, self.d_latent).to(self.device)
# 
434    def z_to_w(self, z: torch.Tensor):
435        w = self.mapping_network(z)
436        return w[:, None, :].expand(-1, self.n_layers, -1)
# 
438    def image_noise(self, batch_size):
439        return torch.rand(batch_size, self.image_size, self.image_size, 1, device=self.device)
# 
441    def generate_images(self, batch_size):
442        z_style = self.z_noise(batch_size)
443        w_style = self.z_to_w(z_style)
444        noise = self.image_noise(batch_size)
445
446        generated_images = self.generator(w_style, noise)
447
448        return generated_images, w_style
# 
450    def step(self, idx):
#

Train Discriminator

452        self.discriminator_optimizer.zero_grad()
453
454        for i in range(self.gradient_accumulate_steps):
455            generated_images, _ = self.generate_images(self.batch_size)
456            fake_output = self.discriminator(generated_images.detach())
457
458            x = next(self.loader).to(self.device)
459            x.requires_grad_()
460            real_output = self.discriminator(x)
461
462            real_loss, fake_loss = self.discriminator_loss(real_output, fake_output)
463            disc_loss = real_loss + fake_loss
464
465            if (idx + 1) % self.lazy_gradient_penalty_interval == 0:
466                gp = self.gradient_penalty(x, real_output)
467                tracker.add('loss.gp', gp)
468                disc_loss = disc_loss + gp
469
470            disc_loss.backward()
471
472            tracker.add('loss.discriminator', disc_loss)
473
474        self.discriminator_optimizer.step()
#

Train Generator & Mapping Network

477        self.generator_optimizer.zero_grad()
478        self.mapping_network_optimizer.zero_grad()
479
480        for i in range(self.gradient_accumulate_steps):
481            generated_images, w_style = self.generate_images(self.batch_size)
482            fake_output = self.discriminator(generated_images)
483
484            gen_loss = self.generator_loss(fake_output)
485
486            if idx > self.lazy_path_penalty_after and (idx + 1) % self.lazy_path_penalty_interval == 0:
487                ppl = self.path_length_penalty(w_style, generated_images)
488                if not torch.isnan(ppl):
489                    gen_loss = gen_loss + ppl
490
491            gen_loss.backward()
492
493            tracker.add('loss.generator', gen_loss)
494
495        self.generator_optimizer.step()
496        self.mapping_network_optimizer.step()
497
498        if (idx + 1) % self.log_generated_interval == 0:
499            tracker.add('generated', generated_images)
500        if (idx + 1) % self.save_checkpoint_interval == 0:
501            experiment.save_checkpoint()
502
503        tracker.save()
# 
505    def run(self):
506        for i in monit.loop(150_000):
507            self.step(i)
508            if (i + 1) % self.log_generated_interval == 0:
509                tracker.new_line()
