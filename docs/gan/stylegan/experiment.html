119    def init(self):
120        self.dataset = Dataset(self.image_size)
121        dataloader = torch.utils.data.DataLoader(self.dataset, batch_size=self.batch_size,
122                                                 num_workers=32,
123                                                 shuffle=True, drop_last=True, pin_memory=True)
124        self.loader = cycle_dataloader(dataloader)
125        log_resolution = int(math.log2(self.image_size))
126
127        self.discriminator = Discriminator(log_resolution).to(self.device)
128        self.generator = Generator(log_resolution, self.d_latent).to(self.device)
129        self.n_gen_blocks = self.generator.n_blocks
130        self.mapping_network = MappingNetwork(self.d_latent, self.mapping_network_layers).to(self.device)
131        self.path_length_penalty = PathLengthPenalty(0.99).to(self.device)
132
133        if self.log_layer_outputs:
134            hook_model_outputs(self.mode, self.discriminator, 'discriminator')
135            hook_model_outputs(self.mode, self.generator, 'generator')
136            hook_model_outputs(self.mode, self.mapping_network, 'mapping_network')
137
138        self.discriminator_loss = DiscriminatorLoss().to(self.device)
139        self.generator_loss = GeneratorLoss().to(self.device)
140
141        self.discriminator_optimizer = torch.optim.Adam(
142            self.discriminator.parameters(),
143            lr=self.learning_rate, betas=self.betas
144        )
145        self.generator_optimizer = torch.optim.Adam(

156    def get_w(self, batch_size: int):
157        if torch.rand(()).item() < self.style_mixing_prob:
158            cross_over_point = int(torch.rand(()).item() * self.n_gen_blocks)
159            z1 = torch.randn(batch_size, self.d_latent).to(self.device)
160            w1 = self.mapping_network(z1)
161            z2 = torch.randn(batch_size, self.d_latent).to(self.device)
162            w2 = self.mapping_network(z2)
163            w1 = w1[None, :, :].expand(cross_over_point, -1, -1)
164            w2 = w2[None, :, :].expand(self.n_gen_blocks - cross_over_point, -1, -1)
165            return torch.cat((w1, w2), dim=0)
166        else:
167            z = torch.randn(batch_size, self.d_latent).to(self.device)
168            w = self.mapping_network(z)
169            return w[None, :, :].expand(self.n_gen_blocks, -1, -1)
# 
171    def image_noise(self, batch_size):
172        noise = []
173        r = 4
174        for i in range(self.n_gen_blocks):
175            if i == 0:
176                n1 = None
177            else:
178                n1 = torch.randn(batch_size, 1, r, r, device=self.device)
179            n2 = torch.randn(batch_size, 1, r, r, device=self.device)
180
181            noise.append((n1, n2))
182            r *= 2
183
184        return noise

186    def generate_images(self, batch_size: int):
187        w_style = self.get_w(batch_size)
188        noise = self.image_noise(batch_size)

190        generated_images = self.generator(w_style, noise)

192        return generated_images, w_style

194    def step(self, idx):
195        with monit.section('Discriminator'):
196            self.discriminator_optimizer.zero_grad()
197
198            for i in range(self.gradient_accumulate_steps):
199                with self.mode.update(is_log_activations=(idx + 1) % self.log_generated_interval == 0):
200                    generated_images, _ = self.generate_images(self.batch_size)
201                    fake_output = self.discriminator(generated_images.detach())
202
203                    real_images = next(self.loader).to(self.device)
204                    if (idx + 1) % self.lazy_gradient_penalty_interval == 0:
205                        real_images.requires_grad_()
206                    real_output = self.discriminator(real_images)
207
208                    real_loss, fake_loss = self.discriminator_loss(real_output, fake_output)
209                    disc_loss = real_loss + fake_loss
210
211                    if (idx + 1) % self.lazy_gradient_penalty_interval == 0:
212                        gp = self.gradient_penalty(real_images, real_output)
213                        tracker.add('loss.gp', gp)
214                        disc_loss = disc_loss + 0.5 * self.gradient_penalty_coefficient * gp * self.lazy_gradient_penalty_interval
215
216                    disc_loss.backward()
217
218                    tracker.add('loss.discriminator', disc_loss)
219
220            if (idx + 1) % self.log_generated_interval == 0:
221                tracker.add('discriminator', self.discriminator)
222            torch.nn.utils.clip_grad_norm_(self.discriminator.parameters(), max_norm=1.0)
223            self.discriminator_optimizer.step()
224
225        with monit.section('Generator'):
226            self.generator_optimizer.zero_grad()
227            self.mapping_network_optimizer.zero_grad()
228
229            for i in range(self.gradient_accumulate_steps):
230                generated_images, w_style = self.generate_images(self.batch_size)
231                fake_output = self.discriminator(generated_images)
232
233                gen_loss = self.generator_loss(fake_output)
234
235                if idx > self.lazy_path_penalty_after and (idx + 1) % self.lazy_path_penalty_interval == 0:
236                    ppl = self.path_length_penalty(w_style, generated_images)
237                    if not torch.isnan(ppl):
238                        tracker.add('loss.ppl', ppl)
239                        gen_loss = gen_loss + ppl
240
241                gen_loss.backward()
242
243                tracker.add('loss.generator', gen_loss)
244
245            if (idx + 1) % self.log_generated_interval == 0:
246                tracker.add('generator', self.generator)
247                tracker.add('mapping_network', self.mapping_network)
248
249            torch.nn.utils.clip_grad_norm_(self.generator.parameters(), max_norm=1.0)
250            torch.nn.utils.clip_grad_norm_(self.mapping_network.parameters(), max_norm=1.0)
251
252            self.generator_optimizer.step()
253            self.mapping_network_optimizer.step()
254
255        if (idx + 1) % self.log_generated_interval == 0:
256            tracker.add('generated', torch.cat([generated_images[:6], real_images[:3]], dim=0))
257        if (idx + 1) % self.save_checkpoint_interval == 0:
258            experiment.save_checkpoint()
259
260        tracker.save()
# 
262    def train(self):
263        for i in monit.loop(150_000):
264            self.step(i)
265            if (i + 1) % self.log_generated_interval == 0:
266                tracker.new_line()
