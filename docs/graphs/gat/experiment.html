57class GAT(Module):
# 
58    def __init__(self, in_features: int, n_hidden: int, n_classes: int, n_heads: int, dropout: float):
59        super().__init__()
60
61        self.layer1 = GraphAttentionLayer(in_features, n_hidden, n_heads, is_concat=True, dropout=dropout)
62
63        self.dropout = nn.Dropout(dropout)
64
65        self.output = GraphAttentionLayer(n_hidden, n_classes, 1, is_concat=False, dropout=dropout)
66
67        self.activation = nn.ELU()
# 
69    def __call__(self, x: torch.Tensor, adj_mat: torch.Tensor):
70        x = self.dropout(x)
71        x = self.layer1(x, adj_mat)
72        x = self.activation(x)
73        x = self.dropout(x)
74        return self.output(x, adj_mat)

81class Configs(BaseConfigs):
82    model: GAT

84    in_features: int

98    def initialize(self):
99        self.cora_dataset = CoraDataset(self.include_edges)
100        self.n_classes = len(self.cora_dataset.classes)
101        self.in_features = self.cora_dataset.features.shape[1]
102        self.model = GAT(self.in_features, self.n_hidden, self.n_classes, self.n_heads, self.dropout).to(self.device)
103        optimizer_conf = OptimizerConfigs()
104        optimizer_conf.parameters = self.model.parameters()
105        self.optimizer = optimizer_conf
# 
107    def run(self):
108        features = self.cora_dataset.features.to(self.device)
109        labels = self.cora_dataset.labels.to(self.device)
110        edges_adj = self.cora_dataset.adj_mat.to(self.device)
111        edges_adj = edges_adj.unsqueeze(-1)
112
113        idx_rand = torch.randperm(len(labels))
114        idx_train = idx_rand[:self.training_samples]
115        idx_valid = idx_rand[self.training_samples:]
116
117        for epoch in monit.loop(self.epochs):
118            self.model.train()
119            self.optimizer.zero_grad()
120            output = self.model(features, edges_adj)
121            loss = self.loss_func(output[idx_train], labels[idx_train])
122            tracker.add('loss.train', loss)
123            tracker.add('accuracy.train', accuracy(output[idx_train], labels[idx_train]))
124            loss.backward()
125            self.optimizer.step()
126
127            self.model.eval()
128
129            with torch.no_grad():
130                output = self.model(features, edges_adj)
131
132                loss = self.loss_func(output[idx_valid], labels[idx_valid])
133                tracker.add('loss.valid', loss)
134                tracker.add('accuracy.valid', accuracy(output[idx_valid], labels[idx_valid]))
135
136            tracker.save()
